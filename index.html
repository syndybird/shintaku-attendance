const SHEET_ID = '1gazIUqo_9yOfsW3PzCZt3eUht4Qo3HC71du7PEGYWuQ'; const SHEET_NAME = '打刻ログ'; const MEMBER_SHEET_NAME = '社員名簿'; function doPost(e) { const lock = LockService.getScriptLock(); try { if (!lock.tryLock(30000)) { return jsonResponse({ status: 'error', message: '混雑しています。しばらく経ってから再度お試しください。' }); } if (!e.postData || !e.postData.contents) { return jsonResponse({ status: 'error', message: 'データが空です。' }); } const params = JSON.parse(e.postData.contents); const action = params.action; const userId = params.userId; const ss = SpreadsheetApp.openById(SHEET_ID); // ユーザー自動登録処理 if (action === 'register') { const name = params.name; if (!userId || !name) { return jsonResponse({ status: 'error', message: 'ユーザーIDまたは名前が取得できませんでした。' }); } let memberSheet = ss.getSheetByName(MEMBER_SHEET_NAME); if (!memberSheet) { memberSheet = ss.insertSheet(MEMBER_SHEET_NAME); memberSheet.appendRow(['ユーザーID', '名前']); } const lastRow = memberSheet.getLastRow(); if (lastRow > 1) { const userIds = memberSheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); if (userIds.includes(userId)) { return jsonResponse({ status: 'error', message: 'このLINEアカウントはすでに登録されています。' }); } } memberSheet.appendRow([userId, name]); return jsonResponse({ status: 'success', message: 'ユーザー登録が完了しました。' }); } // 通常の打刻処理 if (action !== 'attendance') { return jsonResponse({ status: 'ok', message: '待機中' }); } const type = params.type; if (!userId || !type) { return jsonResponse({ status: 'error', message: 'ユーザー情報または打刻区分が取得できませんでした。' }); } // 社員名簿チェック const isAllowed = checkAllowedUser(ss, userId); if (!isAllowed) { return jsonResponse({ status: 'error', message: 'このLINEアカウントは、勤怠システムへの利用が許可されていません。まずは【初期登録用QR】から登録してください。' }); } if (type !== '出勤' && type !== '退勤') { return jsonResponse({ status: 'error', message: '打刻区分が不正です。' }); } let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['日付', '時刻', 'ユーザーID', '打刻区分']); } // 打刻チェック const checkResult = checkLastStatus(sheet, userId, type); if (!checkResult.success) { return jsonResponse({ status: 'error', message: checkResult.message, lastTime: checkResult.lastTime }); } const now = new Date(); recordAttendance(sheet, userId, type, now); const dateTimeStr = Utilities.formatDate(now, 'Asia/Tokyo', "M'月'd'日' HH'時'mm'分'"); return jsonResponse({ status: 'success', message: type + '処理が完了しました。', datetime: dateTimeStr }); } catch (error) { return jsonResponse({ status: 'error', message: 'システムエラーが発生しました: ' + error.toString() }); } finally { lock.releaseLock(); } } function checkAllowedUser(ss, userId) { const memberSheet = ss.getSheetByName(MEMBER_SHEET_NAME); if (!memberSheet) return false; const lastRow = memberSheet.getLastRow(); if (lastRow <= 1) return false; const userIds = memberSheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); return userIds.includes(userId); } function recordAttendance(sheet, userId, type, now) { const date = Utilities.formatDate(now, 'Asia/Tokyo', 'yyyy/MM/dd'); const time = Utilities.formatDate(now, 'Asia/Tokyo', 'HH:mm:ss'); sheet.appendRow([date, time, userId, type]); } function checkLastStatus(sheet, userId, currentType) { const lastRow = sheet.getLastRow(); if (lastRow <= 1) { if (currentType === '退勤') { return { success: false, message: '本日の出勤データが見つかりません。先に「出勤」のQRコードを読み込んでください。', lastTime: null }; } return { success: true }; } const data = sheet.getRange(2, 1, lastRow - 1, 4).getValues(); let lastType = ''; let lastFullDateTimeStr = ''; for (let i = data.length - 1; i >= 0; i--) { if (data[i][2] === userId) { lastType = data[i][3]; const dateObj = new Date(data[i][0]); const timeObj = new Date(data[i][1]); let monthDay = ""; let hourMin = ""; if (!isNaN(dateObj.getTime())) { monthDay = Utilities.formatDate(dateObj, 'Asia/Tokyo', "M'月'd'日'"); } else { const s = data[i][0].toString(); monthDay = s.includes('/') ? `${s.split('/')[1]}月${s.split('/')[2]}日` : s; } if (!isNaN(timeObj.getTime())) { hourMin = Utilities.formatDate(timeObj, 'Asia/Tokyo', 'HH:mm'); } else { hourMin = data[i][1].toString().substring(0, 5); } lastFullDateTimeStr = `${monthDay} ${hourMin}`; break; } } if (currentType === '退勤' && lastType === '') { return { success: false, message: '本日の出勤データが見つかりません。先に「出勤」のQRコードを読み込んでください。', lastTime: null }; } if (currentType === '出勤' && lastType === '出勤') { return { success: false, message: 'すでに本日分の出勤処理は完了しています。連続しての出勤処理はできません。', lastTime: lastFullDateTimeStr }; } if (currentType === '退勤' && lastType === '退勤') { return { success: false, message: 'すでに本日分の退勤処理は完了しています。連続しての退勤処理はできません。', lastTime: lastFullDateTimeStr }; } return { success: true }; } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
